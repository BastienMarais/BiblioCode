Tri par insertions :

Pseudo code :

Fonction TriInsertion(tableau T) :
// T[0] est la sentinelle /!\
// Variables locales : 
// i,j des entiers ;
// X un élément de T ;	
	
	Pour i de 2 à N faire :
		j = i - 1 ;
		X = T[i] ;
		Tant que T[i] > X faire :
			T[j + 1] = T[j] ;
			j = j - 1 ;
		Fin tant que ;
		T[j + 1] = X ;
	Retourner T ;

Fin TriInsertion ;

Exemple python :

def triParInsertion(tab):
	j = 0
	x = 0
	tab.insert(0,min(tab)-1)
	for i in range(2,len(tab)):
		j = i - 1 
		x = tab[i]
		while tab[j] > x :
			tab[j+1] = tab[j]
			j = j - 1
		tab[j+1] = x
	tab.remove(min(tab))
	return tab

Tri Rapide :

Pseudo code :

Fonction TriRapide(tableau T, entier gauche, entier droite) :
// Utilise : Placer, TriRapide ;
// Variables locales : 
// pivot un entier ;

	Si gauche < droite alors :
		T,pivot = Placer(T,gauche,droite) ;
		T = TriRapide(T,gauche,pivot - 1) ;
		T = TriRapide(T,pivot + 1,droite) ;
		Retourner T ;
	Fin si ;
	
Fin TriRapide ;

Fonction Placer(tableau T, entier gauche, entier droite) :
// Utilise Echanger ;
// Variables locales :
// bas, haut et pivot trois entiers ;

	bas = gauche + 1 ;
	haut = droite ;
	Tant que bas <= haut faire :
		Tant que T[bas] <= T[gauche] faire :
			bas = bas +1 ;
		Fin tant que ;
		Tant que T[haut] > T[gauche] faire :
			haut = haut - 1 ;
		Fin tant que ;
		Si bas < haut alors :
			Echanger(T[bas], T[haut]);
			bas = bas + 1 ;
			haut = haut - 1 ;
		Fin si ;
	Fin tant que ;
	
	Echanger(T[gauche], T[haut]) ;
	pivot = haut ;
	Retourner T,pivot ;
	
Fin Placer ;

Fonction Echanger(a, b) :
	Retourner b, a ;
Fin Echanger ;

Exemple python :
